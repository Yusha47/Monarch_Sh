Goldbachs Conjecture

Conjecture: Every even integer greater than 222 can be expressed as the sum of two primes. No proof, that's why it's called a conjecture.

Problem: For all even integers xxx from 444 to 10510^5105, find any two primes such that their sum is xxx. Solve it in less than 1s1s1s.

Bonus: For all odd integers xxx from 444 to 10510^5105, find any three primes such that their sum is xxx. Solve it in less than 1s1s1s.

#include<bits/stdc++.h>
using namespace std;

const int N = 1e5 + 9;

int spf[N];
vector<int> primes;
void sieve() {
  for(int i = 2; i < N; i++) {
    if (spf[i] == 0) spf[i] = i, primes.push_back(i);
    int sz = primes.size();
    for (int j = 0; j < sz && i * primes[j] < N && primes[j] <= spf[i]; j++) {
      spf[i * primes[j]] = primes[j];
    }
  }
}
bool ok(int n) { // checking if possible
  for (int i = 2; i < n; i++) {
    if (spf[i] == i and spf[n - i] == n - i) {
      return true;
    }
  }
  return false;
}
int32_t main() {
  ios_base::sync_with_stdio(0);
  cin.tie(0);
  sieve();
  for (int n = 4; n <= 100000; n += 2) {
    if (!ok(n)) cout << n << '\n';
  }
  return 0;
}

    ll res = 1;
    while (b >= 1) {
        if (b & 1) res = mod_mul(res, a, mod);
        a = mod_mul(a, a, mod);
        b /= 2ll;
    }
    return res;
}
